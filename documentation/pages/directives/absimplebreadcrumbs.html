The breadcrumbs directive is getting the data from the ui-router state configurations. For this page it is:

pageBreadCrumbs: [
    { name: "Home" },
    { name: "Directives" },
    { name: "simple breadcrumb" }
]

Html Markup:

<!-- simple bread crumb directive-->
<ol class="breadcrumb hidden-sm hidden-xs" ab-simple-breadcrumbs>
    <li ng-repeat="item in pageBreadCrumbs" ng-class="{ active: $last }">
        <i ng-if="$first" class="fa fa-dashboard"></i>
        <a href="javascript:">{{ item.name }}</a>
    </li>
</ol>
