Usage

Include the scroll to top markup or htlp file just before the wrapper end tag.


AngularJs Html Markup

<!-- Scroll To Top Component -->
<div class="hidden-xs" ng-include src="'pages/shared/scrolltotop.html'"></div>

Html file content

<!-- Scroll To Top Component -->
<div id="scroll-to-top">
    <a href="javascript:" id="return-to-top">
        <i class="fa fa-chevron-up"></i>
    </a>
</div>

<script>
    $(window).scroll(function () {
        if ($(this).scrollTop() >= 100) {
            $('#return-to-top').fadeIn(400);
        } else {
            $('#return-to-top').fadeOut(400);
        }
    });
    $('#return-to-top').click(function () {
        $('body,html').animate({
            scrollTop: 0
        }, 600);
    });
</script>
